SELECT  Nombre,Unidades,tipobase.Tipo,provebase.Proveedor FROM examen.productos as produc
INNER JOIN examen.tipos as tipobase ON produc.Tipo=tipobase.Id
INNER JOIN examen.proveedores as provebase ON produc.Proveedor=provebase.Id
WHERE tipobase.Tipo="Quesos";



SELECT provebase.Proveedor,SUM(Precio*Unidades) AS Total FROM examen.productos as produc
INNER JOIN examen.proveedores as provebase ON produc.Proveedor=provebase.Id
GROUP BY provebase.Proveedor;


SELECT tipobase.Tipo,SUM(Precio*Unidades) AS Total FROM examen.productos as produc

GROUP BY tipobase.Id;

SELECT  provebase.Proveedor,SUM(Unidades) as Total FROM examen.productos as produc

SELECT  Nombre,(precio*Unidades) AS Ventas,caducidad,Disponible,tipobase.Tipo,provebase.Proveedor FROM examen.productos as produc
INNER JOIN examen.proveedores as provebase ON produc.Proveedor=provebase.Id
INNER JOIN examen.tipos as tipobase ON produc.Tipo=tipobase.Id
