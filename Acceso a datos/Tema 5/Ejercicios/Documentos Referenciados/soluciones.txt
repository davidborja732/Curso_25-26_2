    {

            from: "libros",

    {
        $group:{
            _id:"$usuario_id",
            total_prestamos:{$sum:1},
            prestamos_activos:{
                $sum: {$cond:[{$eq:["$devuelto",false]},1,0]}
            }
        }

        $match:{total_prestamos:{$gt:1}}
    },
    {
        $lookup:{
            from: "usuarios",
            localField: "_id",
            foreignField: "_id",
            as: "usuario"
        }
    },
    {
        $unwind : "$usuario"
    },
    {
        $project:{
            "usuario.nombre":1,
            "usuario.email":1,
            "total_prestamos":1,
            "prestamos_activos":1,
            "_id":0,
        }
    },
    {
        $sort : {total_prestamos:-1},
    }
])
